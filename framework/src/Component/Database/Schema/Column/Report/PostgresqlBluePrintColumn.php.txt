<?php
namespace Laventure\Component\Database\Schema\Column;


use Laventure\Component\Database\Schema\Column\Contract\ColumnInterface;
use Laventure\Component\Database\Schema\Columns\PostgresqlColumn;


/**
 * @PostgresqlBluePrintColumn
*/
class PostgresqlBluePrintColumn extends BluePrintColumn
{

    /**
     * @inheritDoc
    */
    public function add(string $name, string $type, int $length = 11, $default = null, bool $autoincrement = false): ColumnInterface
    {
        $column = new PostgresqlColumn(
            compact('name', 'type', 'length', 'default', 'autoincrement')
        );

        return $this->columns->addColumn($column);
    }




    /**
     * @inheritDoc
    */
    public function increments($name): ColumnInterface
    {
        return $this->add($name, 'serial');
    }



    /**
     * @param $name
     * @return Column
    */
    public function datetime($name): Column
    {
        return $this->add($name, 'timestamp', false);
    }



    /**
     * @param $name
     * @return Column
    */
    public function boolean($name): Column
    {
        return $this->add($name, 'BOOLEAN');
    }



    /**
     * @inheritDoc
    */
    public function printColumns(): string
    {
        //  $this->buildTableColumns()
        return "";
    }
}
