<?php
namespace Laventure\Component\Database\Schema\Column;



/**
 * @MysqlBluePrintColumn
*/
class MysqlBluePrintColumn extends BluePrintColumn
{

    /**
     * @inheritDoc
    */
    public function increments($name): Column
    {
         return $this->add($name, 'int', 11, null, true);
    }



    /**
     * @inheritDoc
    */
    public function add(string $name, string $type, int $length = 11, $default = null, bool $autoincrement = false): Column
    {
        $column = new MysqlColumn(
            compact('name', 'type', 'length', 'default', 'autoincrement')
        );

        return $this->columns->addColumn($column);
    }



    /**
     * @inheritDoc
    */
    public function printColumns(): string
    {
         return "";
    }
}
